(def plus (a b c)
    (~ c (+ a b))
)

(def minus (a b c)
    (if (! a b)
        (~ c (- a b))
    )
)

(def foo (c d)
    (if (! c 0)
        (~ d (+ d 1))
        (~ c (- c 1))
        (foo c d)
    )
)

(def main (a)
    (let q 0)
    (let p 0)
    (plus 45 45 q)
    (minus 30 20 p)
    (foo p q)
    (~ a q)
)
